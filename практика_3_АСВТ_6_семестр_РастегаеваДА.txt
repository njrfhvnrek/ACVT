#include <8051.h>

void delay(unsigned int count)
{
    unsigned int i, j;
    for(i = 0; i < count; i++)
        for(j = 0; j < 100; j++);
}

void lcd_cmd(unsigned char cmd)
{
    P0 = cmd;
    P2 = 0x01;
    P2 = 0x00;
    delay(1);
}

void lcd_data(unsigned char dat)
{
    P0 = dat;
    P2 = 0x03;
    P2 = 0x02;
    delay(1);
}

void lcd_print(char *str)
{
    while(*str)
    {
        lcd_data(*str);
        str++;
    }
}

void lcd_print_sparse(char *str)
{
    while(*str)
    {
        lcd_data(*str);
        lcd_data(' ');
        str++;
    }
}

void lcd_print_repeat(char *str)
{
    unsigned char k;
    for(k = 0; k < 3; k++)
    {
        lcd_print(str);
        lcd_data(' ');
    }
}

void lcd_shift_left(void)
{
    lcd_cmd(0x18);
}

void lcd_clear(void)
{
    lcd_cmd(0x01);
    delay(10);
}

void main(void)
{
    unsigned char i;
    
    unsigned char str1[] = { 'd','a','s','h','a', 0 };
    unsigned char str2[] = { 'd','a','s','h','k','a', 0 };
    
    lcd_cmd(0x38);
    lcd_cmd(0x0E);
    lcd_cmd(0x01);
    delay(10);
    
    lcd_cmd(0xC0);
    lcd_print_sparse(str2);
    
    lcd_cmd(0x80);
    lcd_print_repeat(str1);
    
    while(1)
    {
        for(i = 0; i < 20; i++)
        {
            delay(50);
            lcd_shift_left();
        }
        
        delay(100);
        lcd_clear();
        
        lcd_cmd(0xC0);
        lcd_print_sparse(str2);
        
        lcd_cmd(0x80);
        lcd_print_repeat(str1);
    }
}
