#include <8051.h>

void delay()
{
    unsigned char j;
    for(j=0;j<200;j++);
}

void main()
{
    unsigned char massiv[10] =
    {
        0xC0,  
        0xF9,  
        0xA4,  
        0xB0,  
        0x99,  
        0x92,  
        0x82,  
        0xF8,  
        0x80,  
        0x90   
    };

    unsigned char i = 0;           
    unsigned char j = 1;           
    unsigned char mode = 0;        
    P3 |= 0x06;                    
    P2 = massiv[0];                

    while(1)
    {
        
        if((P3 & 0x02) == 0)
        {
            mode = 0;              
            i = 0;                 
            P2 = massiv[i];        
            delay();               
            while((P3 & 0x02) == 0); 
        }
        
        
        if((P3 & 0x04) == 0)
        {
            mode = 1;              
            j = 1;                 
            P2 = massiv[j];        
            delay();               
            while((P3 & 0x04) == 0);
        }

        
        if(mode == 0)
        {
            P2 = massiv[i];
            delay();
            
            
            i++;
            if(i > 9) i = 0;
        }
        else
        {
            P2 = massiv[j];
            delay();
            
            
            j += 2;
            if(j > 9) j = 1;     
        }
    }
}
