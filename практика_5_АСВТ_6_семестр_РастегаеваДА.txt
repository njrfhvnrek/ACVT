#include <8051.h>

void delay(int x) {
    unsigned int i;
    for (i = 0; i < x; i++) {
        TH0 = 0xD8;
        TL0 = 0xF0;
        TR0 = 1;
        while(TF0 == 0);
        TF0 = 0;
        TR0 = 0;
    }
}

void lcd_command(unsigned char cmd) {
    P0 = cmd;
    P2 = 0x01;
    P2 = 0x00;
    TMOD = 0x01;
    delay(1);
}

void lcd_data(unsigned char dat) {
    P0 = dat;
    P2 = 0x03;
    P2 = 0x02;
    TMOD = 0x01;
    delay(1);
}

void create_smiley(void) {
    unsigned char i;
    unsigned char smiley_pattern[8] = {
        0x00,
        0x0A,
        0x0A,
        0x00,
        0x11,
        0x0E,
        0x00,
        0x00
    };
    
    lcd_command(0x40); // ykazatel adressa CGRAM na nachalo
    for(i = 0; i < 8; i++) {
        lcd_data(smiley_pattern[i]);
    }
    lcd_command(0x80); //ykazatel adressa DDRAM 
}

void main(void) {
    unsigned char row, col, key;
    unsigned char last_key = 0xFF;
    unsigned char press_counter = 0;
    unsigned char current_key = 0xFF;
    unsigned char key_processed = 0;
    unsigned char prev_port_state = 0x0F;
    unsigned int timer_count = 0;
    unsigned char timer_active = 0;
    
    unsigned char keys[4][4] = {
        {'/','=','0','C'},
        {'*','3','2','1'},
        {'-','6','5','4'},
        {'+','9','8','7'}
    };

    TMOD = 0x01;
    
    lcd_command(0x38);
    lcd_command(0x0C);
    lcd_command(0x01);
    delay(1);
    lcd_command(0x06);
    
    create_smiley();

    P1 = 0x0F;

    while(1) {
        unsigned char port_state = P1 & 0x0F;
        
        if(timer_active) { 
            timer_count++;
            if(timer_count >= 30000) { 
                timer_active = 0;
                timer_count = 0;
                press_counter = 0;
                last_key = 0xFF;
            }
        }
        
        if(prev_port_state == 0x0F && port_state != 0x0F) {
            col = 0xFF;
            if((port_state & 0x01) == 0) col = 0;
            else if((port_state & 0x02) == 0) col = 1;
            else if((port_state & 0x04) == 0) col = 2;
            else if((port_state & 0x08) == 0) col = 3;
            
            P1 = 0xF0;
            
            TH0 = 0xFF;
            TL0 = 0xF0;
            TR0 = 1;
            while(TF0 == 0);
            TF0 = 0;
            TR0 = 0;
            
            row = 0xFF;
            if((P1 & 0x10) == 0) row = 0;
            else if((P1 & 0x20) == 0) row = 1;
            else if((P1 & 0x40) == 0) row = 2;
            else if((P1 & 0x80) == 0) row = 3;
            
            P1 = 0x0F;
            
            if(row != 0xFF && col != 0xFF && !key_processed) {
                current_key = keys[row][col];
                
                if(current_key == last_key && timer_active) {
                    press_counter++;
                    if(press_counter >= 5) {
                        lcd_data(0x00);
                        timer_active = 0; // stop timer
                        timer_count = 0; // obnulyaem timer
                        press_counter = 0; // obnulyaem chetchik nazatii
                        last_key = 0xFF; // cbrasivaem last clavishy
                        key_processed = 1; // ystanavlivaem flag obrabotki nazatiy. poka 0 - yderzanie ne vizovet povtornix vivodov
                    }
                } else {
                    lcd_data(current_key);
                    last_key = current_key;
                    press_counter = 1; // first nazatie v new seria
                    timer_active = 1; 
                    timer_count = 0;
                    key_processed = 1;
                }
            }
        }
        else if(prev_port_state != 0x0F && port_state == 0x0F) {
            TH0 = 0xFE;
            TL0 = 0x0C;
            TR0 = 1;
            while(TF0 == 0);
            TF0 = 0;
            TR0 = 0;
            
            if((P1 & 0x0F) == 0x0F) {
                key_processed = 0;
            }
        }
        
        prev_port_state = port_state;
        
        TH0 = 0xFF;
        TL0 = 0xCE;
        TR0 = 1;
        while(TF0 == 0);
        TF0 = 0;
        TR0 = 0;
    }
}
