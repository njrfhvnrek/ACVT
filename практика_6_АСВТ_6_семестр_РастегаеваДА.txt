#include <8051.h>

void uart_send(unsigned char byte)
{
    SBUF = byte;
    while(!TI);
    TI = 0;
}

void main(void)
{
    unsigned char i;
    unsigned char code data_array[20] = {
        0x30, 0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37, 0x38, 0x39,
        0x3A, 0x3B, 0x3C, 0x3D, 0x3E, 0x3F, 0x40, 0x41, 0x42, 0x43
    };
    
    SCON = 0x50;
    PCON = 0x00;
    TMOD &= 0x0F;
    TMOD |= 0x20;
    TH1 = 0xFA;
    TL1 = 0xFA;
    TR1 = 1;
    
    for(i = 0; i < 100; i++);
    
    for(i = 0; i < 20; i++)
    {
        uart_send(data_array[i]);
    }
    
    while(1);
}
